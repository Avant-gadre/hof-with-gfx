on_actions = {
	on_weekly = {
		effect = {
			JAP = {
				JAP_interservice_swap = yes
			}
		}
	}
	on_daily_JAP = {
		effect = {
			JAP_interDiff_calculation = yes
		}
	}
	on_startup = {
		effect = {
			JAP = {
				JAP_lock_all_initial_templates = yes
			}
		}
	}
	on_new_term_election = {
		random_events = {
			100 = rw_jap_election.1
			100 = rw_jap_election.2
			100 = rw_jap_election.3
			100 = rw_jap_election.4
			100 = rw_jap_election.5
		}
	}
	on_puppet = {
		effect = {
			if = {
				limit = {
					original_tag = BEY
					FROM = {
						tag = JAP
					}
				}
				set_cosmetic_tag = BEY_JAP
			}
			if = {
				limit = {
					original_tag = CHI
					FROM = {
						tag = JAP
					}
				}
				set_cosmetic_tag = CHI_JAP
			}
		}
	}
	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					original_tag = BEY
					FROM = {
						tag = JAP
					}
				}
				set_cosmetic_tag = BEY_JAP
			}
			if = {
				limit = {
					original_tag = CHI
					FROM = {
						tag = JAP
					}
				}
				set_cosmetic_tag = CHI_JAP
			}
		}
	}
}
