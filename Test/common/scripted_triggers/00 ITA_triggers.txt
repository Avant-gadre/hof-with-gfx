ENG_tyrant_possibly_alliance = {
	ENG = {
		exists = yes
		NOT = {
			has_war_with = ROOT
		}
	}
	OR = {
		AND = {
			ENG = {
				is_faction_leader = yes
			}
			ROOT = {
				is_in_faction = no
			}
		}
		AND = {
			ROOT = {
				is_faction_leader = yes
			}
			ENG = {
				is_in_faction = no
			}
		}
		AND = {
			ROOT = {
				is_in_faction = no
			}
			ENG = {
				is_in_faction = no
			}
		}
	}
	ENG = {
		has_government = fascism
	}
	ROOT = {
		has_government = fascism
	}
}

ITA_can_attack_other_tier1_trigger = {
	OR = {
		AND = {
			ENG = {
				NOT = {
					is_in_faction_with = ROOT
				}
			}
			OR = {
				threat > 0.3
				date > 1938.1.1
			}
		}
		AND = {
			is_in_faction_with = ENG
			ENG = {
				has_government = neutrality
				is_subject = no
			}
			threat > 0.5
		}
	}
}
