SPA_spainish_civil_war_ongoing_trigger = {
	custom_trigger_tooltip = {
		tooltip = SPA_spainish_civil_war_ongoing_tt
		has_global_flag = Spain_civil_war_start_flag
		NOT = {
			has_global_flag = Spain_civil_war_end_flag
		}
	}
}

	}
}
