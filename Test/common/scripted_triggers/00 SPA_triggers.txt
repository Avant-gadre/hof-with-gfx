SPA_pl_spoe_coalition_trigger = {
	has_country_flag = SPA_psoe_join_election_flag
	if = {
		limit = {
			OR = {
				is_ai = no
				NOT = {
					has_country_flag = SPA_i_want_to_cheat_flag
				}
			}
		}
		# Liberal-PSOE Coalition
		set_temp_variable = {
			var = SPA_pl_spoe_coalition_var
			value = SPA_psoe_influcence
		}
		add_to_temp_variable = {
			var = SPA_pl_spoe_coalition_var
			value = SPA_liberal_influence
		}
		if = {
			limit = {
				has_country_flag = SPA_psoe_split_flag
			}
			set_temp_variable = {
				var = SPA_psoe_modified_var
				value = SPA_psoe_influcence
			}
			multiply_temp_variable = {
				var = SPA_psoe_modified_var
				value = -0.5
			}
			add_to_temp_variable = {
				var = SPA_pl_spoe_coalition_var
				value = SPA_psoe_modified_var
			}
		}
		if = {
			limit = {
				has_country_flag = SPA_psoe_join_revolution_flag
			}
			set_temp_variable = {
				var = SPA_psoe_modified_var
				value = SPA_psoe_influcence
			}
			multiply_temp_variable = {
				var = SPA_psoe_modified_var
				value = -1
			}
			add_to_temp_variable = {
				var = SPA_pl_spoe_coalition_var
				value = SPA_psoe_modified_var
			}
		}
		# Liberal-CEDA Coalition
		set_temp_variable = {
			var = SPA_pl_ceda_coalition_var
			value = SPA_ceda_influence
		}
		add_to_temp_variable = {
			var = SPA_pl_ceda_coalition_var
			value = SPA_ceda_influence
		}
		# CEDA-Rights Coalition
		set_temp_variable = {
			var = SPA_ceda_right_coalition_var
			value = SPA_ceda_influence
		}
		add_to_temp_variable = {
			var = SPA_ceda_right_coalition_var
			value = SPA_carlist_influence
		}
		# Comparisons
		OR = {
			check_variable = {
				SPA_pl_spoe_coalition_var > SPA_pl_ceda_coalition_var
			}
			check_variable = {
				SPA_pl_spoe_coalition_var = SPA_pl_ceda_coalition_var
			}
		}
		OR = {
			check_variable = {
				SPA_pl_spoe_coalition_var > SPA_ceda_right_coalition_var
			}
			check_variable = {
				SPA_pl_spoe_coalition_var = SPA_ceda_right_coalition_var
			}
		}
	}
}
