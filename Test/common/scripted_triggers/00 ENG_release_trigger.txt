ENG_is_east_indies_states_trigger = {
	OR = {
		is_core_of = MAL
		is_core_of = INS
	}
	NOT = {
		state = 830
		state = 721
	}
}

ENG_JAP_control_all_east_indies_states_trigger = {
	hidden_trigger = {
		all_state = {
			OR = {
				AND = {
					ENG_is_east_indies_states_trigger = yes
					OR = {
						THIS.controller = {
							OR = {
								OR = {
									tag = ENG
									is_puppet_of = ENG
									is_in_faction_with = ENG
								}
								OR = {
									tag = JAP
									is_puppet_of = JAP
									is_in_faction_with = JAP
								}
							}
						}
					}
				}
				ENG_is_east_indies_states_trigger = no
			}
		}
	}
}
