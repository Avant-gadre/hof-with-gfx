is_enemy_ideology = {
	OR = {
		AND = {
			has_government = communism
			ROOT = {
				OR = {
					has_government = fascism
					has_government = democratic
				}
			}
		}
		AND = {
			has_government = democratic
			ROOT = {
				OR = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		AND = {
			has_government = fascism
			ROOT = {
				OR = {
					has_government = democratic
					has_government = communism
				}
			}
		}
	}
}

has_liberal_government = {
	custom_trigger_tooltip = {
		tooltip = has_liberal_government_tt
		OR = {
			has_global_flag = Show_all_advisor_flag
			AND = {
				has_government = democratic
				has_conservative_government = no
			}
			has_country_flag = has_liberal_government_flag
		}
	}
}

has_socialist_government = {
	custom_trigger_tooltip = {
		tooltip = has_socialist_government_tt
		OR = {
			has_global_flag = Show_all_advisor_flag
			has_government = communism
			has_country_flag = has_socialist_government_flag
		}
	}
}

has_dictatorship_government = {
	custom_trigger_tooltip = {
		tooltip = has_dictatorship_government_tt
		OR = {
			has_global_flag = Show_all_advisor_flag
			has_country_flag = has_dictatorship_government_flag
			has_government = fascism
			AND = {
				has_government = neutrality
				has_conservative_government = no
			}
		}
	}
}

has_conservative_government = {
	custom_trigger_tooltip = {
		tooltip = has_conservative_government_tt
		OR = {
			has_country_flag = has_conservative_government_flag
			has_global_flag = Show_all_advisor_flag
			has_country_leader_ideology = classical_liberism
			has_country_leader_ideology = oligarchism
			has_country_leader_ideology = conservatism
			has_country_leader_ideology = authoritarianism
		}
	}
}

has_valkism_government = {
	custom_trigger_tooltip = {
		tooltip = has_valkism_government_tt
		has_country_leader_ideology = valkism
	}
}

has_elected_socialist_government = {
	custom_trigger_tooltip = {
		tooltip = has_elected_socialist_government_tt
		OR = {
			has_global_flag = Show_all_advisor_flag
			AND = {
				has_elections = yes
				has_government = communism
			}
		}
	}
}

has_right_wing_government = {
	OR = {
		has_conservative_government = yes
		has_dictatorship_government = yes
	}
}

has_left_wing_government = {
	OR = {
		has_socialist_government = yes
		has_liberal_government = yes
	}
}

has_liberal_conservative_government = {
	OR = {
		has_liberal_government = yes
		has_conservative_government = yes
	}
}

has_non_dictatorship_government = {
	OR = {
		has_elected_socialist_government = yes
		has_liberal_government = yes
		has_conservative_government = yes
	}
}
