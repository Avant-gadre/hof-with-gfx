is_enemy_ideology = {
	OR = {
		AND = {
			has_government = communism
			ROOT = {
				OR = {
					has_government = fascism
					has_government = democratic
				}
			}
		}
		AND = {
			has_government = democratic
			ROOT = {
				OR = {
					has_government = communism
					has_government = fascism
				}
			}
		}
		AND = {
			has_government = fascism
			ROOT = {
				OR = {
					has_government = democratic
					has_government = communism
				}
			}
		}
	}
}

has_liberal_government = {
	custom_trigger_tooltip = {
		tooltip = has_a_liberal_government_tt
		OR = {
			##has_country_leader_ideology = social_democracy
			has_country_leader_ideology = liberalism
			has_country_leader_ideology = classical_liberalism
			has_country_leader_ideology = moderatism
			has_country_leader_ideology = authoritarianism
		}
	}
}

has_socialist_government = {
	custom_trigger_tooltip = {
		tooltip = has_a_socialist_government_tt
		OR = {
			has_government = communism
			has_country_leader_ideology = social_democracy
		}
	}
}

has_dictatorship_government = {
	custom_trigger_tooltip = {
		tooltip = has_a_dictatorship_government_tt
		OR = {
			has_government = fascism
			has_country_leader_ideology = despotism
			has_country_leader_ideology = stratocracy
			has_country_leader_ideology = absolute_monarchy
			has_country_leader_ideology = reactionary
		}
	}
}
