DIPLOMACY_INCOMING_LEND_LEASE_ENABLE_TRIGGER = {
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = FREE
			}
		}
		always = yes
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = LIMITED
			}
		}
		always = yes		#handled by game logic, see above
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = SAME_IDEOLOGY
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_LEND_LEASE_SAME_IDEOLOGY_TT
			has_government = FROM
		}
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = SAME_FACTION
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_LEND_LEASE_SAME_FACTION_TT
			is_in_faction_with = FROM
		}
	}
	if = {
		limit = {
			has_game_rule = {
				rule = allow_lend_lease
				option = BLOCKED
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_LEND_LEASE_BLOCKED_TT
			always = no
		}
	}
}

game_weight_can_add_fascist = {
	GER = {
		AND = {
			OR = {
				check_variable = {
					fascist_game_weight < democratic_game_weight
				}
				check_variable = {
					fascist_game_weight = democratic_game_weight
				}
			}
		}
		AND = {
			OR = {
				check_variable = {
					fascist_game_weight < unaligned_game_weight
				}
				check_variable = {
					fascist_game_weight = unaligned_game_weight
				}
			}
		}
		AND = {
			OR = {
				check_variable = {
					fascist_game_weight < communist_game_weight
				}
				check_variable = {
					fascist_game_weight = communist_game_weight
				}
			}
		}
	}
}

DIPLOMACY_JOIN_ALLY_ENABLE_TRIGGER = {
	if = {
		limit = {
			ROOT = {
				is_faction_leader = yes
			}
		}
		always = yes
	}
	else_if = {
		limit = {
			ROOT = {
				is_faction_leader = no
			}
			FROM = {
				is_faction_leader = yes
			}
		}
		always = yes
	}
	else = {
		custom_trigger_tooltip = {
			tooltip = DIPLOMACY_JOIN_ALLY_ENABLE_TRIGGER_tt
			always = no
		}
	}
}
