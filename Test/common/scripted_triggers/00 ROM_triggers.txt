ROM_control_all_intital_romania_states_trigger = {
	hidden_trigger = {
		all_state = {
			OR = {
				AND = {
					ROM_is_intital_romania_states_trigger = yes
					is_fully_controlled_by_ROOT_or_subject = yes
				}
				ROM_is_intital_romania_states_trigger = no
			}
		}
	}
}
