#scripted trigger for Hungary
is_available_fighter_ROM = {
	OR = {
		AND = {
			has_tech = fighter1
			ROM = {
				NOT = {
					has_tech = fighter1
				}
			}
		}
		AND = {
			has_tech = fighter2
			ROM = {
				NOT = {
					has_tech = fighter2
				}
			}
		}
		AND = {
			has_tech = fighter3
			ROM = {
				NOT = {
					has_tech = fighter3
				}
			}
		}
	}
	NOT = {
		has_war_with = ROM
	}
}

ROM_is_transilvania_state_trigger = {
	OR = {
		state = 76
		state = 82
		state = 83
		state = 789
		state = 84
	}
}

ROM_control_all_transilvania_state_trigger = {
	custom_trigger_tooltip = {
		tooltip = ROM_control_all_transilvania_state_tt
		all_state = {
			OR = {
				AND = {
					ROM_is_transilvania_state_trigger = yes
					is_fully_controlled_by_ROOT_or_subject = yes
				}
				ROM_is_transilvania_state_trigger = no
			}
		}
	}
}

			has_completed_focus = ROM_constitutional_guarantees
		}
	}
}
