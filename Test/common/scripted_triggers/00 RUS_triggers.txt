#scripted trigger for RUS
is_available_fighter_RUS = {
	OR = {
		AND = {
			has_tech = fighter1
			RUS = {
				NOT = {
					has_tech = fighter1
				}
			}
		}
		AND = {
			has_tech = fighter2
			RUS = {
				NOT = {
					has_tech = fighter2
				}
			}
		}
		AND = {
			has_tech = fighter3
			RUS = {
				NOT = {
					has_tech = fighter3
				}
			}
		}
	}
	NOT = {
		has_war_with = RUS
	}
}

RUS_has_tie_with_japan_trigger = {
	custom_trigger_tooltip = {
		tooltip = RUS_has_tie_with_japan_tt
		NOT = {
			has_war_with = JAP
			has_completed_focus = RUS_break_tie_with_jap
		}
		OR = {
			has_completed_focus = RUS_expand_japanese_advisory_group
			is_subject_of = JAP
		}
		OR = {
			is_in_faction_with = JAP
			has_idea = RUS_Japanese_economic_control_1_idea
			has_idea = RUS_Japanese_economic_control_2_idea
			has_idea = RUS_Japanese_economic_control_3_idea
			has_idea = RUS_Japanese_economic_control_4_idea
		}
	}
}
