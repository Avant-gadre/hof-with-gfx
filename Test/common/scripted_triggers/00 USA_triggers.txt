congress_low_support_trigger = {
	custom_trigger_tooltip = {
		tooltip = congress_low_support_trigger_tt
		check_variable = {
			var = senators_support
			value = quorum_low
			compare = greater_than_or_equals
		}
		check_variable = {
			var = representatives_support
			value = 174
			compare = greater_than_or_equals
		}
	}
}
