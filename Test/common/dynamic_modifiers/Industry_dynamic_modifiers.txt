agrarian_state = {
	enable = {
		always = yes
	}
	remove_trigger = {
		OR = {
			AND = {
				infrastructure > 4
				OR = {
					compliance > 40
					is_core_of = ROOT
				}
			}
			has_dynamic_modifier = {
				modifier = half_industrial_state
			}
			has_dynamic_modifier = {
				modifier = industrial_area_Lv1
			}
			has_dynamic_modifier = {
				modifier = industrial_area_Lv2
			}
			has_dynamic_modifier = {
				modifier = industrial_area_Lv3
			}
			has_dynamic_modifier = {
				modifier = industrial_area_Lv4
			}
		}
	}
	icon = GFX_HOF_state_modifier_agriculture
	state_production_speed_rail_way_factor = -0.4
	state_production_speed_infrastructure_factor = -0.4
	state_production_speed_buildings_factor = -0.2
	recruitable_population_factor = -0.4
	army_speed_factor = -0.4
	monthly_population = 0.4
}
