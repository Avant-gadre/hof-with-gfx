ENG_hold_speech_effect = {
	add_stability = 0.05
	add_war_support = 0.05
	if = {
		limit = {
			USA = {
				exists = yes
				OR = {
					is_in_faction_with = ROOT
					has_war = no
				}
			}
			ENG = {
				NOT = {
					has_government = fascism
				}
			}
		}
		USA = {
			add_war_support = 0.1
		}
	}
}
