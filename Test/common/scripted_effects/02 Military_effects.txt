add_one_motor_or_mechanize_bonus = {
	if = {
		limit = {
			NOT = {
				has_tech = motorised_infantry
			}
		}
		add_tech_bonus = {
			name = motorised_infantry
			bonus = 3
			uses = 1
			category = motorized_equipment
		}
	}
	else = {
		add_tech_bonus = {
			name = mechanized
			bonus = 3
			uses = 1
			category = cat_mechanized_equipment
		}
	}
}
