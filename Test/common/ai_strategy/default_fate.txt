		num_divisions < 90
	}

		type = railway_gun_divisions_ratio
		value = 4
	}
}

default_produce_floating_harbors = {

	ai_strategy = {
		type = unit_ratio
		id = floating_harbor		# for floating harbors, 'unit_ratio' is actually NOT a ratio but an absolute desired number
		value = 10
	}
}
