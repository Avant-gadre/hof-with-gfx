				factor = 0.5

				factor = 2
				is_subject_of = ITA
			}
		}
	}

		name = fate_roman_diplo.1.b
		trigger = {
		}
		GRE = {
			add_ideas = ITA_economy_integration_idea
		}
		ai_chance = {
			base = 1
		}
	}
	##We refuse

		name = fate_roman_diplo.1.c
		trigger = {
			##always = no
		}

		add_political_power = -30

			country_event = {
				id = fate_roman_diplo.2
				days = 2
			}
		}
		ai_chance = {
			base = 0
		}
	}
}
