remove_claim_and_add_core_and_transfer_of_ROOT = {
	if = {
		limit = {
			is_claimed_by = ROOT
		}
		remove_claim_by = ROOT
	}
	if = {
		limit = {
			NOT = {
				is_core_of = ROOT
			}
		}
		add_core_of = ROOT
	}
	if = {
		limit = {
			NOT = {
				is_owned_by = ROOT
			}
		}
		transfer_state_to = ROOT
	}
}
