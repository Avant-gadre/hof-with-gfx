add_research_slot_until_six = {
	if = {
		limit = {
			amount_research_slots < 6
		}
		add_research_slot = 1
	}
	else_if = {
		limit = {
			has_idea = generic_research_advances
		}
		modify_timed_idea = {
			idea = generic_research_advances
			days = 730
		}
	}
	else = {
		add_timed_idea = {
			idea = generic_research_advances
			days = 730
		}
	}
}

remove_fort_to_level_1 = {
	set_building_level = {
		type = bunker
		level = 1
		province = {
			all_provinces = yes
			limit_to_border = no
			level > 1
		}
	}
}
