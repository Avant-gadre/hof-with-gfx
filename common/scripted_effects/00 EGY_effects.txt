EGY_guardian_of_africa_effect = {
	every_country = {
		limit = {
			THIS.capital_scope = {
				is_on_continent = africa
			}
			has_capitulated = no
			is_independent_and_samefaction_and_exist_and_peace_off_ROOT = yes
		}
		add_opinion_modifier = {
			target = THIS
			modifier = generic_friendly_opinion
		}
	}
}
