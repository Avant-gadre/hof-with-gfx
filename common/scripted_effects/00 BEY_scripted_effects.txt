	}
	else_if = {
		limit = {

		}
		swap_ideas = {

			add_idea = BEY_army_corruption_4_idea
		}
	}
	else_if = {
		limit = {
			has_idea = BEY_army_corruption_4_idea
		}
		swap_ideas = {
			remove_idea = BEY_army_corruption_4_idea
