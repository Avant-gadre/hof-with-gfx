BEY_down_army_corruption_effect = {
	if = {
		limit = {
			has_idea = BEY_army_corruption_1_idea
		}
		swap_ideas = {
			remove_idea = BEY_army_corruption_1_idea
			add_idea = BEY_army_corruption_1_idea
		}
	}
	else_if = {
		limit = {
			has_idea = BEY_army_corruption_2_idea
		}
		swap_ideas = {
			remove_idea = BEY_army_corruption_2_idea
			add_idea = BEY_army_corruption_3_idea
		}
	}
	else_if = {
		limit = {
			has_idea = BEY_army_corruption_3_idea
		}
		remove_ideas = BEY_army_corruption_3_idea
	}
}

		limit = {
			has_country_flag = BEY_priorize_fengtian_flag
		}
		random_owned_controlled_state = {
			if = {
				limit = {
					BEY_is_fengtian_states_trigger = yes
				}
				add_one_civilian_factory = yes
			}
