on_actions = {
	on_startup = {
		effect = {
			set_global_flag = allowrenaming_flag
			every_country = {
				limit = {
					exists = yes
				}
				every_controlled_state = {
					rename_state_effect = yes
				}
			}
		}
	}
	on_government_change = {
		effect = {

			if = {
				limit = {
					has_global_flag = allowrenaming_flag
				}
				every_controlled_state = {
					rename_state_effect = yes
				}
			}
		}
	}
	on_state_control_changed = {
		effect = {

			### Endonym Check
			if = {
				limit = {
					has_global_flag = allowrenaming_flag
				}
				FROM.FROM = {
					rename_state_effect = yes
				}
			}
		}
	}
}
