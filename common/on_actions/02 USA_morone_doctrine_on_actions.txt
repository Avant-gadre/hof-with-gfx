on_actions = {
	##门罗主义
	on_declare_war = {
		effect = {
			if = {
				limit = {
					OR = {
						FROM = {
							tag = USA
						}
						tag = USA
					}
				}
				##这是美国的敌人
				every_country = {
					limit = {
						OR = {
							any_allied_country = {
								has_war_with = USA
							}
							has_war_with = USA
						}
					}
					##这是美国的小弟
					every_country = {
						if = {
							limit = {
								NOT = {
									has_war_with = USA
								}
								any_allied_country = {
									NOT = {
										has_war_with = USA
									}
								}
								has_idea = USA_morone_doctrine_idea
							}
							embargo_PREV = yes
							set_country_flag = monroe_doctrine_embragoed@THIS
							PREV = {
								set_country_flag = monroe_doctrine_embragoed@THIS
							}
						}
					}
				}
			}
		}
	}
