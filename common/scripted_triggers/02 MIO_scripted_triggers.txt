is_British_Sphere = {
	custom_trigger_tooltip = {
		tooltip = is_British_Sphere_tt
		FROM = {
			is_British_Sphere_tags = yes
			OR = {
				is_subject = no
				is_subject_of = ENG
			}
		}
		OR = {
			FROM = {
				original_tag = ENG
			}
			FROM = {
				is_ally_with = ENG
				OR = {
					original_tag = MLT
					original_tag = KEN
					original_tag = SAF
					original_tag = RAJ
					original_tag = BUR
					original_tag = MAL
					original_tag = INS
					original_tag = AST
					original_tag = BAH
					original_tag = CAN
				}
			}
			FROM = {
				original_tag = POL
				is_ally_with = ENG
				has_completed_focus = POL_align_with_the_west
			}
			FROM = {
				original_tag = IRQ
				NOT = {
					has_war_with = ENG
				}
			}
		}
	}
}
