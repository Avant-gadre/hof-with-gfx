##皂盒决议判定
is_owned_or_controlled_by_ROOT_or_subject = {
	OR = {
		controller = {
			OR = {
				tag = ROOT
				is_subject_of = ROOT
			}
		}
		owner = {
			OR = {
				tag = ROOT
				is_subject_of = ROOT
			}
		}
	}
}

is_controlled_by_ROOT_or_ally = {
	controller = {
		is_ally_with = ROOT
	}
}

HUN_target_can_Take_over_land = {
	OR = {
		is_subject_of = HUN
		is_in_faction_with = HUN
	}
	NOT = { tag = HUN }
	NOT = {
		OR = {
			tag = ROM
			tag = SLV
			tag = SER
			tag = YUG
			tag = CZE
		}
		is_subject = no
	}
}
