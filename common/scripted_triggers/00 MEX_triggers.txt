ENG_possibly_alliance_MEX = {
	ENG = {
		exists = yes
		NOT = {
			has_war_with = ROOT
		}
	}
	has_government = neutrality
	OR = {
		AND = {
			ENG = {
				is_faction_leader = yes
			}
			ROOT = {
				is_in_faction = no
			}
		}
		AND = {
			ROOT = {
				is_faction_leader = yes
			}
			ENG = {
				is_in_faction = no
			}
		}
		AND = {
			ROOT = {
				is_in_faction = no
			}
			ENG = {
				is_in_faction = no
			}
		}
	}
	OR = {
		ENG = {
			has_government = neutrality
		}
		ENG = {
			has_government = democratic
		}
	}
}
