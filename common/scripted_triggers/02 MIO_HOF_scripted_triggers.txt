is_Japan_Sphere = {
	AND = {
		is_Japan_Sphere_tags = yes
		OR = {
			FROM = {
				is_subject = no
			}
			FROM = {
				is_subject_of = JAP
			}
		}
		OR = {
			FROM = {
				original_tag = JAP
			}
			AND = {
				OR = {
					FROM = {
						OR = {
							original_tag = PHI
							original_tag = SRP
							original_tag = KOR
							original_tag = MAN
							original_tag = INI
							original_tag = INA
						}
						is_ally_with = JAP
					}
				}
			}
			AND = {
				FROM = {
					original_tag = RUS
					AND = {
						OR = {
							has_completed_focus = rus_join_jap_faction
							has_completed_focus = RUS_Toward_Japan
						}
						OR = {
							has_idea = RUS_Japanese_economic_control_1_idea
							has_idea = RUS_Japanese_economic_control_2_idea
							has_idea = RUS_Japanese_economic_control_3_idea
							has_idea = RUS_Japanese_economic_control_4_idea
						}
						is_ally_with = JAP
					}
				}
			}
		}
	}
}
