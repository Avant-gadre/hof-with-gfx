		cost = 100
		fire_only_once = yes
		visible = {
			has_completed_focus = ENG_appeal_to_imperial_loyalists

			}
		}

		complete_effect = {
			custom_effect_tooltip = ENG_renew_support_for_imperialist_coup_in_canada_tt

				set_country_flag = ENG_prepare_coup_in_CAN
			}

					id = mtg_britain.61
