	is_triggered_only = yes
	
	option = { # Ja
		name = swe_mot.3.a

			base = 10
			modifier = {

                    FROM = { has_government = fascism }
                    ROOT = { has_government = democratic }
                }
				factor = 0								
			}
		}

			country_event = { days = 1 id = swe_mot.1 }
		}

		name = swe_mot.3.b

			base = 0
