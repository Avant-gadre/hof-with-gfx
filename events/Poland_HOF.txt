##### 1936 Polish Election
country_event = {
	id = poland_hof.1
	title = poland_hof.1.t
	desc = poland_hof.1.d
	picture = GFX_report_event_pol_1936_election
	
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			set_country_flag = POL_election_triggered_flag
		}
	}
	
	option = {
		name = poland_hof.1.a
		### PSL Elected
		trigger = {
			check_variable = {
				var = POL_PSL_Power
				value = POL_Sanacja_Power
				compare = greater_than_or_equals
			}
			check_variable = {
				var = POL_PSL_Power
				value = POL_Endecja_Power
				compare = greater_than_or_equals
			}
		}

	}

		name = poland_hof.1.b

		add_stability = -0.10

			ideology = democratic
			popularity = 0.05
		}

			limit = {
				has_country_flag = POL_PSL_PPS_deal_flag

			limit = {

					var = Sanacja_Repression
					value = -19
					compare = less_than
				}
			}
			hidden_effect = {
				country_event = { id = poland_hof.9 days = 6 }

			}

			### Repression failed, PSL made a government
		}

			limit = { 
				set_temp_variable = { Sanacja_Repression = POL_Sanacja_repression }
				subtract_from_temp_variable = { Sanacja_Repression = POL_PSL_Power }

					var = Sanacja_Repression
					value = -20
					compare = greater_than
				}

					var = Sanacja_Repression

			hidden_effect = {

				country_event = { id = poland_hof.11 days = 10 }
