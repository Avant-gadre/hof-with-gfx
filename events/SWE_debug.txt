	title = swe_mot.27.t
	desc = swe_mot.27.d
	picture = GFX_report_event_generic_scathach_skadi

	fire_only_once = yes
	
	trigger = {
		tag = SWE
    is_subject = no
    or = {
      DEN = { exists = no }
      DEN = { is_subject_of = GER }
		  GER = { 
        has_full_control_of_state = 99 
        has_full_control_of_state = 37
      }
    }
    or = {
      NOT = { owns_state = 100 }
      NOT = { owns_state = 101 }
      NOT = { owns_state = 337 }
    }
    GER = { 
      has_completed_focus = GER_Control_Skagerrak_strait
      has_full_control_of_state = 99 
    }
  }

	mean_time_to_happen = {
		days = 3
	}
	
	option = {
		name = swe_mot.27.a

		ai_chance = {
			base = 100
		}

    add_political_power = -50
    transfer_state = 100
    transfer_state = 101
    transfer_state = 337
	}
    
  option = {
		name = swe_mot.26.b
	}
