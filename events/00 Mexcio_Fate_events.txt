country_event = {
	id = gsk_mexico.6
	title = gsk_mexico.6.t
	desc = gsk_mexico.6.d
	picture = GFX_report_event_quetzalcoatl

	is_triggered_only = yes
	
	option = {
		name = gsk_mexico.6.a
        effect_tooltip = {
			if = {
				limit = {
				    ROOT = { is_faction_leader = yes }
				}
				add_to_faction = MEX
				}
			else_if= {
					limit = { MEX = { is_faction_leader = yes } }
					MEX = { add_to_faction = ROOT }
					else = { 
						MEX = {
							set_rule = { can_create_factions = yes }
							create_faction = Tenochtitlan_treaty
							add_to_faction = ROOT
						}      
					}
				}		
		
			}
		FROM = {
			add_ai_strategy = {
				type = alliance
				id = "ROOT"
				value = 200
			}
		}
		add_ai_strategy = {
				type = alliance
				id = "MEX"
				value = 200
			}
		FROM = { country_event = { id =  gsk_mexico.7 } }
		hidden_effect = {
			
		}
		}
		

	
	option = {
		name = gsk_mexico.6.b
		ai_chance = { factor = 0 }
		FROM = { 

			country_event = { id = gsk_mexico.8 }
		}
		
	}
}
